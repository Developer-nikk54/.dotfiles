# tmux shorctus :

# daily use manage split-pane/window
  mouse = right click 
  split hr = prefix => "
  split vr = prefix => %
  reload tmux = prefix => r
  window overview = prefix => w
  switch window = prefix => alt+1,2,3....
  rename window = prefix => ,

# tmux session manage
  create new session = prefix => N
  close current active session = prefix => q
  session overview = prefix => s
  switch session = prefix => ( )
  rename session = prefix => $

# copy mode keybinds
  copy mode enable = prefix => [
  select text = space or V
  copy text = Enter
  paste text = prefix => ]
  search (down/up) = ? or /
  jump (forward/backward) = (f/F) or (t/T)

# others keybind
  see tmux keybind = prefix => ?
  attach pop-up terminal = prefix => ctrl-p
  detach pop-up terminal = prefix => d
